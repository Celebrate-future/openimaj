30  package org.openimaj.logger;

32  import org.apache.logging.log4j.Level;
33  import org.apache.logging.log4j.LogManager;
34  import org.apache.logging.log4j.Logger;
35  import org.apache.logging.log4j.core.Filter.Result;
36  import org.apache.logging.log4j.core.appender.ConsoleAppender;
37  import org.apache.logging.log4j.core.filter.ThresholdFilter;
38  import org.apache.logging.log4j.core.layout.PatternLayout;
39  
40  /**
41   * Make using log4j slightly less awful
42   *
43   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
44   *
45   */
46  public class LoggerUtils {
47  	/**
48  	 * Prepare a console logger with a default layout. This allows {@link Logger}
49  	 * instances to be used while still being useful as the line to prepare logger
50  	 * can be removed and the default log4j.properties will be correctly listened
51  	 * to.
52  	 */
53  	public static void prepareConsoleLogger() {
54  		final String PATTERN = "%m\n";
55  		final PatternLayout layout = PatternLayout.newBuilder().withPattern(PATTERN).build();
56  		final ThresholdFilter filter = ThresholdFilter.createFilter(Level.DEBUG, Result.ACCEPT, Result.DENY);
57  
58  		final ConsoleAppender console = ConsoleAppender.newBuilder().withName("default-consoleappender")
59  				.withLayout(layout).withFilter(filter).build();
60  
61  		// add appender to any Logger (here is root)
62  		final Logger root = LogManager.getRootLogger();
63  		if (root instanceof org.apache.logging.log4j.core.Logger)
64  			((org.apache.logging.log4j.core.Logger) root).addAppender(console);
65  	}
66  
67  	/**
68  	 * Debug message occationally
69  	 *
70  	 * @param logger
71  	 * @param message
72  	 * @param b
73  	 */
74  	public static void debug(Logger logger, String message, boolean b) {
75  		if (b) {
76  			logger.debug(message);
77  		}
78  	}
79  
80  	/**
81  	 * Calls {@link #format(Logger, String, Level, Object...)} with level
82  	 * {@link Level#DEBUG}
83  	 *
84  	 * @param logger
85  	 * @param string
86  	 * @param obj
87  	 */
88  	public static void debugFormat(Logger logger, String string, Object... obj) {
89  		format(logger, string, Level.DEBUG, obj);
90  	}
91  
92  	/**
93  	 * Checks the level, if acceptable calls
94  	 * {@link String#format(String, Object...)} at the appropriate level
95  	 *
96  	 * @param logger
97  	 * @param string
98  	 * @param debug
99  	 * @param obj
100 	 */
101 	public static void format(Logger logger, String string, Level debug, Object... obj) {
102 		final Level l = logger.getLevel();
103 		if (debug.isMoreSpecificThan(l)) {
104 			logger.log(debug, String.format(string, obj));
105 		}
106 	}
